\begin{frame}{Model problem for the morning}
\small
Let $\Omega\subset\R^2$ have boundary $\Gamma$ and let
\[
  G(\x,\y)=-\frac{1}{2\pi}\log|\x-\y|
\]
be the Laplace Green's function.  The double layer potential is
\[
  \Dop[\mu](\x)
  =\int_\Gamma \frac{\partial G}{\partial \nu_{\y}}(\x,\y)
       \mu(\y)\dd s_{\y}.
\]

\vspace{2mm}
For a smooth boundary and a Dirichlet problem, a standard second-kind equation is
\[
  \left(-\frac12 I + K\right)\mu=f,\qquad
  K\mu(\x)=\operatorname{p.v.}\!\int_\Gamma
    \frac{\partial G}{\partial \nu_{\y}}(\x,\y)\mu(\y)\dd s_{\y}.
\]

\begin{block}{Numerical objective}
Discretize the equation and evaluate the potential with high order accuracy, even when $\x$ is on or very close to $\Gamma$.
\end{block}
\end{frame}

%================================================================
\begin{frame}{Where plain Gauss quadrature breaks}
\small
On one panel $\Gamma_\ell$, parameterized by $\y_\ell(t)$, $t\in[-1,1]$,
\[
  \int_{\Gamma_\ell} K(\x,\y)\mu(\y)\dd s_{\y}
  =
  \int_{-1}^1 K(\x,\y_\ell(t))\mu(\y_\ell(t))|\y_\ell'(t)|\dd t.
\]

\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.52\textwidth}
\begin{itemize}
  \item If $\dist(\x,\Gamma_\ell)\sim h_\ell$, the integrand is resolved by panel Gauss rules.
  \item If $\dist(\x,\Gamma_\ell)\ll h_\ell$, the integrand has a sharp peak.
  \item If $\x\in\Gamma_\ell$, the rule must handle a singular limit, principal value, or jump term.
\end{itemize}
\end{column}

\begin{frame}{Hour I: kernel-split quadrature objectives}
\small
By the break, the main mathematical objectives are:
\begin{enumerate}
  \item state the Laplace layer potential representation in real and complex notation;
  \item explain why near-boundary evaluation has larger quadrature constants than far evaluation;
  \item derive the Cauchy-moment recurrence behind kernel-split quadrature;
  \item identify where jump terms, residues, and branch choices enter;
\end{enumerate}

\vspace{2mm}
The main mathematical transformation is:
\[
  \text{singular quadrature problem}
  \quad\longrightarrow\quad
  \text{analytic contour integral plus smooth interpolation}.
\]
\end{frame}

%================================================================
\begin{frame}{Boundary integral equation setting}
\small
For an elliptic boundary value problem, first choose a layer-potential
representation
\[
  u(\x)=(\mathcal{L}\rho)(\x),\qquad \x\in\Omega .
\]
Then enforce the boundary condition by taking traces:
\[
  B\mathcal{L}\rho=f
  \quad\Longrightarrow\quad
  ((I+\lambda K)\rho)(\x)=h(\x),\qquad \x\in\Gamma .
\]

\begin{itemize}
  \item $\rho$ is an auxiliary density chosen to satisfy the boundary data.
  \item $K$ comes from a boundary trace or normal derivative of the representation.
  \item $K$ is often weakly singular or principal-value singular.
  \item For smooth $\Gamma$, a good second-kind formulation stays well conditioned under refinement.
  \item Once $\rho$ is known, field values require evaluating $\mathcal{L}\rho$ at target points.
\end{itemize}

\begin{block}{Where quadrature enters}
Quadrature appears in both matrix assembly and field evaluation.  The kernels
are related by the representation and trace operations, but they need not be
identical.
\end{block}
\end{frame}

%================================================================
\begin{frame}{Laplace Green's function}
\small
In two dimensions,
\[
  G(\x,\y)=-\frac{1}{2\pi}\log|\x-\y|,
  \qquad
  -\Delta_{\x}G(\x,\y)=\delta(\x-\y).
\]

The singularity is only logarithmic, but its derivatives have Cauchy-type singularities:
\[
  \nabla_{\y}G(\x,\y)
  =
  -\frac{1}{2\pi}\frac{\y-\x}{|\y-\x|^2}.
\]

\vspace{2mm}
For a unit normal $\bnu_{\y}$ at the source point,
\[
  \frac{\partial G}{\partial \nu_{\y}}(\x,\y)
  =
  -\frac{1}{2\pi}
  \frac{(\y-\x)\cdot\bnu_{\y}}{|\y-\x|^2}.
\]
\end{frame}

%================================================================
\begin{frame}{Single and double layer potentials}
\small
\[
  \Sop[\sigma](\x)
  =
  \int_\Gamma G(\x,\y)\sigma(\y)\dd s_{\y},
  \qquad
  \Dop[\mu](\x)
  =
  \int_\Gamma
  \frac{\partial G}{\partial \nu_{\y}}(\x,\y)\mu(\y)\dd s_{\y}.
\]

\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.48\textwidth}
\textbf{Single layer $\Sop$}
\begin{itemize}
  \item logarithmic kernel;
  \item potential is continuous;
  \item normal derivative jumps.
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Double layer $\Dop$}
\begin{itemize}
  \item source-normal derivative of $G$;
  \item potential jumps;
  \item boundary trace gives a second-kind Dirichlet equation.
\end{itemize}
\end{column}
\end{columns}

\vspace{3mm}
The double layer is the clearest setting for the kernel-split contour
quadrature mechanism.
\end{frame}

%================================================================
\begin{frame}{Second-kind equations and conditioning}
\small
A first-kind equation has the form
\[
  \Sop\sigma=f.
\]
After discretization, the matrix becomes increasingly ill-conditioned as $h\to0$.

\vspace{2mm}
A second-kind equation has the form
\[
  \left(\pm\frac12 I+K\right)\mu=f.
\]
For a smooth boundary, the compact boundary operator $K$ does not destroy conditioning under refinement.

\begin{block}{Numerical consequence}
High-order quadrature is useful only if the linear algebra remains stable enough to expose the accuracy.
\end{block}
\end{frame}

%================================================================
\begin{frame}{Computational role of the jump relation}
\small
Let $\Dop[\mu]$ be the double layer potential on a smooth boundary.
As $\x$ approaches $x_0\in\Gamma$ from the two sides,
\[
  \gamma^{\rm ext}\Dop[\mu](x_0)=
  \left(\frac12 I+K\right)\mu(x_0),
  \qquad
  \gamma^{\rm int}\Dop[\mu](x_0)=
  \left(-\frac12 I+K\right)\mu(x_0).
\]

\begin{itemize}
  \item The singular part is local and universal.
  \item The boundary operator $K$ is what must be discretized accurately.
  \item A special quadrature rule must reproduce this jump, not smooth it away.
\end{itemize}
\end{frame}

%================================================================
\begin{frame}{Panelization}
\small
Write the boundary as a union of panels,
\[
  \Gamma=\bigcup_{\ell=1}^{N_{\rm pan}}\Gamma_\ell,\qquad
  \Gamma_\ell=\{\gamma_\ell(t):-1\le t\le1\}.
\]
Then a layer potential becomes
\[
  \sum_{\ell=1}^{N_{\rm pan}}
  \int_{-1}^1
  K(\x,\gamma_\ell(t))\,\rho(\gamma_\ell(t))\,|\gamma_\ell'(t)|\dd t.
\]

\begin{itemize}
  \item Geometry is local.
  \item Refinement can be local.
  \item Quadrature rules can be chosen panel by panel.
\end{itemize}
\end{frame}

%================================================================
\begin{frame}{Composite Gauss-Legendre rule}
\small
On a panel, let $\{t_j,w_j\}_{j=1}^n$ be the $n$-point Gauss-Legendre rule on $[-1,1]$.
\[
  \int_{\Gamma_\ell} f(\y)\dd s_{\y}
  \approx
  \sum_{j=1}^n
  f(\gamma_\ell(t_j))|\gamma_\ell'(t_j)|w_j.
\]

\begin{itemize}
  \item For smooth integrands, this is spectrally accurate in $n$.
  \item Typical high-order panel codes use $n=12,16,$ or $24$.
  \item The problem is not the rule itself; the problem is loss of smoothness in the integrand.
\end{itemize}
\end{frame}

%================================================================
\begin{frame}{Matrix entries for a Nystr\"om method}
\small
For target node $x_i$ and source node $y_j=\gamma_{p_j}(t_j)$, the ordinary
Nystr\"om entry is
\[
  A_{ij}
  =
  \delta_{ij}
  +\lambda
  K(x_i,y_j)\,|\gamma'_{p_j}(t_j)|w_j
\]
only for far/smooth source panels.

Replace or correct this formula for:
\begin{itemize}
  \item boundary traces on a source panel containing the target: jump, diagonal
  limit, or principal-value/log treatment;
  \item close panels, including endpoint neighbors:
  $\dist(x_i,\Gamma_{p_j})/h_{p_j}\le c_{\rm near}$;
  \item off-grid field targets: apply the near-zone test to the representation kernel.
\end{itemize}

\begin{block}{Important distinction}
The criterion is operator- and tolerance-dependent; panel adjacency is only a
starting point for the near-panel search.
\end{block}
\end{frame}

%================================================================
\begin{frame}{Smooth-panel error estimate}
\small
If $F(t)$ is analytic in a complex neighborhood of $[-1,1]$, then Gauss-Legendre quadrature converges rapidly:
\[
  \left|
  \int_{-1}^1F(t)\dd t-\sum_{j=1}^n w_jF(t_j)
  \right|
  \lesssim C\varrho^{-2n}
\]
for some $\varrho>1$ determined by the nearest complex singularity.

\vspace{2mm}
For a close target,
\[
  F(t)=K(z,\gamma(t))\rho(\gamma(t))|\gamma'(t)|
\]
has a nearby complex singularity where $\gamma(t)=z$.
The effective $\varrho$ can be close to one, so rapid convergence is lost.
\end{frame}

%================================================================
\begin{frame}{A local way to see near singularity}
\small
Near a target $z$, approximate a panel by a tangent line:
\[
  \gamma(t)\approx \gamma(t_0)+h(t-t_0)T_0.
\]
Then the Cauchy-like denominator behaves like
\[
  \frac{1}{\gamma(t)-z}
  \approx
  \frac{1}{h(t-t_0)+\ii d},
\]
where $d$ is the signed distance from $z$ to the panel.

\vspace{2mm}
The length scale of variation is
\[
  |t-t_0|\sim \frac{|d|}{h}.
\]
If $|d|\ll h$, the peak is much narrower than the panel.
\end{frame}

%================================================================
\begin{frame}{Classification used by a code}
\small
For each target $z$ and panel $\Gamma_\ell$:
\[
  \eta_\ell(z)=\frac{\dist(z,\Gamma_\ell)}{h_\ell}.
\]

\begin{tabular}{@{}lll@{}}
\toprule
Regime & Typical condition & Quadrature \\
\midrule
far & $\eta_\ell(z)>c_1$ & ordinary Gauss \\
near & $c_0<\eta_\ell(z)\le c_1$ & kernel-split / close rule \\
self or nearly self & $\eta_\ell(z)\le c_0$ & principal value, jump, residue \\
\bottomrule
\end{tabular}

\vspace{3mm}
The constants are empirical and depend on panel order, geometry quality, and requested accuracy.
\end{frame}

\section{Kernel-Split Quadrature}

\begin{frame}{Kernel-split quadrature: the plan}
\small
We must evaluate the double layer potential
\[
  \Dop[\mu](\x)=\int_\Gamma\frac{\partial G}{\partial\nu_{\y}}(\x,\y)\,\mu(\y)\dd s_{\y}
\]
when the target $\x$ is \emph{on} or \emph{close to} $\Gamma$, where ordinary
panel Gauss quadrature loses accuracy.

\vspace{1mm}
Kernel-split quadrature does this in three moves:
\begin{enumerate}
  \item rewrite the real line integral as a complex \hl{contour integral} with
  an explicit Cauchy kernel $1/(\tau-z)$;
  \item interpolate the \emph{smooth} density by a polynomial on each panel;
  \item integrate the singular factor $1/(\tau-z)$ \emph{analytically}, panel by
  panel, through Cauchy moments.
\end{enumerate}

\vspace{1mm}
The only quantity ever approximated is the smooth density.  Everything singular
is done in closed form.
\end{frame}

%================================================================
\begin{frame}{Complex notation and orientation}
\small
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.6\textwidth}
Identify $\x=(x_1,x_2)$ with $z=x_1+\ii x_2$ and a source point with $\tau$.  For
two real vectors written as complex numbers,
\[
  a\cdot b=\operatorname{Re}\{a\bar b\}.
\]
On a \hl{positively oriented} (counterclockwise) boundary, the complex line
element is $\dd\tau=T\dd s$ with unit tangent $|T|=1$.  The outward unit normal
is $T$ rotated clockwise by $90^\circ$:
\[
  n=-\ii T
  \quad\Longleftrightarrow\quad
  T=\ii n
  \quad\Longrightarrow\quad
  n\dd s=-\ii\,\dd\tau.
\]
This orientation choice fixes every sign and jump that follows.
\end{column}
\begin{column}{0.38\textwidth}
\begin{center}
\begin{tikzpicture}[scale=0.9,>=Latex]

\end{frame}

\begin{frame}{The double layer as a Cauchy integral}
\small
The source-normal derivative of $G=-\tfrac{1}{2\pi}\log|\cdot|$ is
\[
  \frac{\partial G}{\partial\nu_\tau}(z,\tau)
  =-\frac{1}{2\pi}\,\frac{(\tau-z)\cdot n}{|\tau-z|^2}
  =-\frac{1}{2\pi}\operatorname{Re}\!\left\{\frac{(\tau-z)\bar n}{|\tau-z|^2}\right\}.
\]
Since $\dfrac{(\tau-z)\bar n}{|\tau-z|^2}=\dfrac{\bar n}{\overline{(\tau-z)}}
=\overline{\left(\dfrac{n}{\tau-z}\right)}$, the real part simplifies:
\[
  \frac{\partial G}{\partial\nu_\tau}(z,\tau)
  =-\frac{1}{2\pi}\operatorname{Re}\!\left\{\frac{n}{\tau-z}\right\}.
\]
Multiply by $\dd s$ and use $n\dd s=-\ii\,\dd\tau$, together with
$\operatorname{Re}\{-\ii w\}=\operatorname{Im}\{w\}$:
\[
  \frac{\partial G}{\partial\nu_\tau}\dd s
  =-\frac{1}{2\pi}\operatorname{Re}\!\left\{\frac{-\ii\,\dd\tau}{\tau-z}\right\}
  =-\frac{1}{2\pi}\operatorname{Im}\!\left\{\frac{\dd\tau}{\tau-z}\right\}.
\]
\[
  \boxed{\;
  \Dop[\mu](z)
  =-\frac{1}{2\pi}\operatorname{Im}\int_\Gamma\frac{\mu(\tau)}{\tau-z}\dd\tau
  \;}
\]
The difficult kernel is now the Cauchy kernel $1/(\tau-z)$; the geometry sits in
$\dd\tau$.
\end{frame}

%================================================================
\begin{frame}{Boundary values and the jump}
\small
Write the Cauchy integral
\[
  \mathcal C\mu(z)=\frac{1}{2\pi\ii}\int_\Gamma\frac{\mu(\tau)}{\tau-z}\dd\tau,
  \qquad\text{so that}\qquad
  \Dop[\mu](z)=-\operatorname{Re}\{\mathcal C\mu(z)\}.
\]
As $z\to z_0\in\Gamma$, the Plemelj--Sokhotski formula gives one-sided limits
\[
  \mathcal C^{\pm}\mu(z_0)=\pm\tfrac12\,\mu(z_0)
  +\operatorname{p.v.}\frac{1}{2\pi\ii}\int_\Gamma\frac{\mu(\tau)}{\tau-z_0}\dd\tau,
\]
where $+$ is the limit from the interior (left of the oriented curve) and $-$
from the exterior.  Hence, with $K\mu=-\operatorname{Re}\{\text{p.v.}\,\mathcal C\mu\}$,
\[
  \Dop^{\rm int}[\mu]=\bigl(-\tfrac12 I+K\bigr)\mu,
  \qquad
  \Dop^{\rm ext}[\mu]=\bigl(+\tfrac12 I+K\bigr)\mu,
  \qquad
  \Dop^{\rm ext}-\Dop^{\rm int}=\mu.
\]
\begin{block}{The numerical point}
The $\pm\tfrac12$ jump is not added by hand: it appears automatically as the
\emph{branch} of the base Cauchy moment, chosen from the side of $\Gamma$ on
which $z$ lies.
\end{block}
\end{frame}

%================================================================
\begin{frame}{One panel, rescaled}
\small
Split $\Gamma=\bigcup_\ell\Gamma_\ell$.  On a panel with endpoints $a,b$, set the
center $z_c=\tfrac{a+b}{2}$ and half-length $\Delta=\tfrac{b-a}{2}$, and scale
\[
  s=\frac{\tau-z_c}{\Delta}\in[-1,1],
  \qquad
  \hat z=\frac{z-z_c}{\Delta},
  \qquad
  \dd\tau=\Delta\,\dd s.
\]
The panel contribution to the double layer is
\[
  \Dop_\ell[\mu](z)
  =-\frac{1}{2\pi}\operatorname{Im}\int_{\Gamma_\ell}\frac{\mu(\tau)}{\tau-z}\dd\tau
  =-\frac{1}{2\pi}\operatorname{Im}\int_{-1}^{1}\frac{\mu(s)}{s-\hat z}\dd s,
\]
where $\Delta$ cancels because $\tau-z=\Delta(s-\hat z)$.  On the panel $\mu$ is
smooth; all the (near-)singular behavior is in $1/(s-\hat z)$.
\end{frame}

%================================================================
\begin{frame}{Interpolate the density, integrate the kernel exactly}
\small
Expand the smooth density in Legendre polynomials of the scaled variable,
\[
  \mu(\tau)\approx\sum_{k=0}^{p-1}c_k\,P_k(s).
\]
With the $p$ Gauss nodes $s_j$ and the Legendre--Vandermonde matrix
$V_{jk}=P_k(s_j)$, the coefficients come from the nodal values by
\[
  c=U\mu_{\rm node},
  \qquad
  U=V^{-1}
\]
--- an $\cO(p^3)$ precomputation \emph{per panel}, independent of the target.
Then
\[
  \Dop_\ell[\mu](z)\approx
  -\frac{1}{2\pi}\operatorname{Im}\sum_{k=0}^{p-1}c_k\,f_k(\hat z),
  \qquad
  f_k(\hat z)=\int_{-1}^{1}\frac{P_k(s)}{s-\hat z}\dd s.
\]
The \hl{Cauchy moments} $f_k$ hold everything target-dependent; the density
enters only through $c$.
\end{frame}

%================================================================
\begin{frame}{Why the moments are computable}
\small
Each Legendre polynomial $P_k(s)$ is \emph{entire}, so $P_k(s)/(s-\hat z)$ is
analytic except at the simple pole $s=\hat z$.  Consequently
\[
  f_k(\hat z)=\int_{-1}^{1}\frac{P_k(s)}{s-\hat z}\dd s
\]
can be deformed off the real axis and evaluated in closed form.

\begin{itemize}
  \item Ordinary Gauss quadrature fails only because the integrand has a pole
  \emph{near} $[-1,1]$; the moment integrates the pole analytically instead.
  \item The value depends on which side of $[-1,1]$ the pole $\hat z$ lies ---
  this choice of branch is exactly the source of the jump relation.
\end{itemize}

\vspace{1mm}
So the accuracy of the panel value is limited only by the polynomial
interpolation of $\mu$, never by the distance from $z$ to the panel.
\end{frame}

%================================================================
\begin{frame}{Cauchy moments by recurrence}
\small
The base moment is a difference of logarithms,
\[
  f_0(\hat z)=\int_{-1}^{1}\frac{\dd s}{s-\hat z}
  =\log(1-\hat z)-\log(-1-\hat z)
  =\log\frac{1-\hat z}{-1-\hat z}
  \quad(\text{branch below}).
\]
The next moment follows from $\dfrac{s}{s-\hat z}=1+\dfrac{\hat z}{s-\hat z}$ and
$\int_{-1}^1\dd s=2$:
\[
  f_1(\hat z)=2+\hat z\,f_0(\hat z).
\]
The Legendre three-term recurrence $(2k+1)sP_k=(k+1)P_{k+1}+kP_{k-1}$, together
with $\int_{-1}^1 P_k\dd s=0$ for $k\ge1$, gives
\[
  f_{k+1}(\hat z)=\frac{(2k+1)\,\hat z\,f_k(\hat z)-k\,f_{k-1}(\hat z)}{k+1},
  \qquad k\ge1.
\]
All $p$ moments cost $\cO(p)$ per target.
\end{frame}

%================================================================
\begin{frame}{The base moment carries the jump}
\small
$f_0=\log\dfrac{1-\hat z}{-1-\hat z}$ is multivalued; the correct value is set by
the side of the panel on which $z$ lies.

\vspace{1mm}
\textbf{Off-surface target.}  Take the nearest source node $\tau_\star$ with
normal $n_\star$ and test the sign of
\[
  \cos\theta=\operatorname{Re}\{(z-\tau_\star)\bar n_\star\}=(z-\tau_\star)\cdot n_\star,
\]
then choose $\gamma=-\ii$ on the exterior side ($\cos\theta>0$) and
$\gamma=+\ii$ on the interior side ($\cos\theta<0$), and set
\[
  f_0=\log\gamma+\log\frac{1-\hat z}{\gamma(-1-\hat z)},
  \qquad \log\gamma=\pm\tfrac{\ii\pi}{2}.
\]
A curved panel needs a convex/concave correction using the panel's interpolated
depth near $\hat z$.

\vspace{1mm}
\textbf{On-surface target (self).}  $f_0=\log(1-\hat z)-\log(-1-\hat z)$ with a
$\pm\ii\pi$ correction near the endpoints: that $\pm\ii\pi$ \emph{is} the
$\pm\tfrac12$ double-layer jump.
\end{frame}

%================================================================
\begin{frame}{Far value, near value, and the correction}
\small
Let $w^{c}_j=\gamma_\ell'(t_j)w_j$ be the complex weights ($\dd\tau$ at node
$j$).  The two panel values are
\[
  \Dop_\ell^{\rm Gauss}(z)=-\frac{1}{2\pi}\operatorname{Im}\sum_{j=1}^{p}\mu_j\,
  \frac{w^{c}_j}{\tau_j-z}
  \qquad(\text{ordinary rule}),
\]
\[
  \Dop_\ell^{\rm split}(z)=-\frac{1}{2\pi}\operatorname{Im}\Bigl[(f\,U)\,\mu_{\rm node}\Bigr],
  \qquad f=[\,f_0,\ldots,f_{p-1}\,].
\]
For a \emph{far} panel the two agree.  For a \emph{near} or \emph{self} panel,
keep the far-field (Gauss/FMM) sum and add only the correction
\[
  \Delta_\ell(z)=\Dop_\ell^{\rm split}(z)-\Dop_\ell^{\rm Gauss}(z).
\]
Storing the per-node correction weights
$\bigl[(fU)_j-w^{c}_j/(\tau_j-z)\bigr]$ makes this a sparse, FMM-compatible
correction to the smooth sum.
\end{frame}

%================================================================
\begin{frame}{Algorithm: close evaluation of the Laplace double layer}
\footnotesize
\textbf{Precompute (once per source panel).}  Complex nodes $\tau_j$, normals
$n_j$, complex weights $w^c_j=\gamma_\ell'(t_j)w_j$, arc-length weights
$w^s_j=|\gamma_\ell'(t_j)|w_j$, and $U=V^{-1}$ with $V_{jk}=P_k(s_j)$.

\medskip
\textbf{Far field.}
$\displaystyle
\Dop_{\rm far}(z)=-\frac{1}{2\pi}\sum_{\text{far }j}\mu_j\operatorname{Im}\!\Bigl\{\frac{w^c_j}{\tau_j-z}\Bigr\}$
\; (accelerated by the FMM).

\medskip
\textbf{For each near or self panel of the target $z$:}
\begin{enumerate}
  \item scale: $\hat z=(z-z_c)/\Delta$;
  \item side test $\to$ branch $\gamma$ (or the self $\pm\ii\pi$ rule);
  \item $f_0$ from the branched logarithm, $f_1=2+\hat z f_0$, then the
  recurrence for $f_2,\ldots,f_{p-1}$;
  \item accurate weights $fU$; correction
  $\Delta_\ell(z)=-\tfrac{1}{2\pi}\operatorname{Im}\bigl[(fU-\text{Gauss})\,\mu_{\rm node}\bigr]$;
  \item add $\Delta_\ell(z)$ to $\Dop_{\rm far}(z)$.
\end{enumerate}

\medskip
\textbf{Cost.}  $\cO(p^3)$ precompute per panel, $\cO(p)$ per (target, near
panel), far field $\cO(N)$ via the FMM.
\end{frame}

%================================================================
\begin{frame}{Two cheap validations}
\small
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.5\textwidth}
\textbf{Unit-circle Cauchy test.}  For $\mu\equiv1$,
\[
  \int_{|\tau|=1}\frac{\dd\tau}{\tau-z}=
  \begin{cases}2\pi\ii,&|z|<1,\\[2pt]0,&|z|>1.\end{cases}
\]
Checks orientation, the residue sign, the branch, and inside/outside
classification in one shot.
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Normal-line jump test.}  With $z_\epsilon^\pm=z_0\pm\epsilon\, n(z_0)$,
\[
  \Dop[\mu](z_\epsilon^+)-\Dop[\mu](z_\epsilon^-)\xrightarrow[\epsilon\to0^+]{}\mu(z_0).
\]
Prescribing $\mu$ separates the quadrature error from any linear-solve error.
\end{column}
\end{columns}

\vspace{2mm}
Predicted behavior: ordinary Gauss loses digits as $z$ approaches $\Gamma$;
kernel-split stays flat to machine precision, with the correct jump across the
boundary.
\end{frame}

%================================================================
\begin{frame}{Single layer: logarithmic moments}
\small
The single layer has the logarithmic kernel
\[
  \Sop[\sigma](z)=-\frac{1}{2\pi}\int_\Gamma\log|\tau-z|\,\sigma(\tau)\dd s
  =-\frac{1}{2\pi}\int_\Gamma\operatorname{Re}\{\log(\tau-z)\}\,\sigma(\tau)\dd s.
\]
The same interpolation reduces it to \hl{log moments}
\[
  q_k(\hat z)=\int_{-1}^{1}P_k(s)\log(s-\hat z)\dd s .
\]
Integration by parts (the endpoint terms cancel because $P_{k\pm1}(\pm1)$ match)
gives a recurrence tied to the Cauchy moments,
\[
  q_0=\log|(1-\hat z)(1+\hat z)|-\hat z\,f_0-2,
  \qquad
  q_k=\frac{f_{k-1}-f_{k+1}}{2k+1}\ \ (k\ge1).
\]
The \emph{imaginary} part of the base log moment does not affect $\Sop$: it is
annihilated by the real arc-length weights, so any convenient branch may be used
for $q_0$.
\end{frame}

%================================================================
\begin{frame}{Other potentials and other equations}
\small
\textbf{Higher singularities.}  Normal derivatives ($\Dpop$, $\Top$) and the
Stokes/elasticity kernels bring $1/(\tau-z)^m$.  Their moments come from
differentiating the Cauchy moments in $\hat z$,
\[
  f_k^{(m)}(\hat z)=\int_{-1}^1\frac{P_k(s)}{(s-\hat z)^{m}}\dd s,
  \qquad
  f_k^{(m+1)}=\frac1m\frac{\dd}{\dd\hat z}f_k^{(m)}\ (+\text{ endpoint terms}),
\]
built by the same kind of recurrence as $f_k$.

\vspace{1mm}
\textbf{Helmholtz.}  The kernels are Bessel functions times the Laplace
singularity.  For the single layer,
$\tfrac{\ii}{4}H_0^{(1)}(k|\tau-z|)$, only the $Y_0\sim\tfrac{2}{\pi}\log$ part
is singular, so the close correction is $-\tfrac1\pi J_0(k|\tau-z|)\times(\text{log-moment correction})$;
the double-layer analog combines the log-moment and Cauchy-moment corrections.

\begin{block}{General pattern}
Every 2D layer potential is a sum of a smooth part (ordinary Gauss) and singular
factors $1/(\tau-z)^m$ and $\log(\tau-z)$, each integrated by its analytic
moment.  Laplace, Helmholtz, Stokes, and elasticity differ only in which moments
appear.
\end{block}
\end{frame}

%================================================================
\begin{frame}{Why it converges}
\small
On a near or self panel, the split rule replaces the smooth density by its
degree-$(p-1)$ interpolant and integrates the singular factor exactly:
\[
  \int_{-1}^1\frac{\mu(s)-\mu_{p-1}(s)}{s-\hat z}\dd s .
\]
\begin{itemize}
  \item The error is the \emph{interpolation} error of $\mu$, not a quadrature
  error for a peaked integrand.
  \item For analytic $\mu$ it decays geometrically, $\|\mu-\mu_{p-1}\|\lesssim
  C\varrho^{-p}$, with $\varrho>1$ independent of the target distance.
  \item The target distance affects only the stability constants, handled by the
  moment recurrence and the branch rule.
\end{itemize}
Ordinary Gauss, by contrast, must resolve a pole at distance $\sim|d|/h$ from
$[-1,1]$, and its error grows without bound as $z\to\Gamma$.
\end{frame}

\section{RCIP}

%================================================================
\begin{frame}{Point singularities in second-kind equations}
\small
Consider a Fredholm second-kind equation
\[
  (I+\lambda K)\rho(r)=\lambda g(r),\qquad r\in\Gamma,
\]
with $\Gamma$ a piecewise smooth closed contour, $\rho$ the unknown layer
density, $g$ smooth data, and $\lambda$ a parameter.  Away from a finite set of
singular points $\gamma_j\in\Gamma$ the operator $K$ is compact.

\begin{itemize}
  \item On a smooth boundary $K$ is compact: its spectrum is discrete, bounded
  by one in modulus, and accumulates only at zero.
  \item At a corner, an endpoint, a triple junction, or a change of boundary
  condition, $K$ loses compactness and its kernel becomes more singular.
  \item The density $\rho$ is then generally singular at each $\gamma_j$.
\end{itemize}

\vspace{1mm}
The singularity is \emph{local}, but if handled naively it contaminates the
entire solve.
\end{frame}

%================================================================
\begin{frame}{Why a corner is hard}
\small
\begin{itemize}
  \item The precise singularity of $\rho$ at $\gamma_j$ is difficult to analyze,
  and depends on the opening angle, the type of junction, and the boundary
  condition.
  \item Resolving $\rho$ by polynomials needs many panels, refined dyadically
  toward $\gamma_j$.
  \item Standard Nystr\"om discretization captures the $L^\infty$ behavior of
  $\rho$, whereas $\rho$ may be \emph{unbounded} at the corner.
  \item Refinement spans many length scales, causing \hl{artificial
  ill-conditioning} and a loss of achievable accuracy in quantities computed
  from $\rho$.
\end{itemize}

\begin{block}{Goal}
Resolve the local singularity without exposing its fine-scale degrees of freedom
to the global linear system.
\end{block}
\end{frame}

%================================================================
\begin{frame}{Three ways to cope with a singular density}
\small
\begin{enumerate}
  \item \textbf{Tailored discretization.}  Build special interpolation nodes and
  generalized Gaussian quadratures for the singular density.  Very accurate, but
  each case (angle, junction type, boundary condition) needs its own rule.
  \item \textbf{Graded mesh.}  Refine dyadically toward $\gamma_j$ and integrate
  a \emph{smooth} density on each panel with a high-order rule.  Needs
  $\ell^2$-type weighting to fight ill-conditioning, and the system grows with
  the refinement.
  \item \textbf{RCIP.}  Change the unknown so that it becomes piecewise smooth,
  compress the local refinement into a small correction, and keep the global
  solve on the coarse mesh.
\end{enumerate}

\vspace{1mm}
\begin{block}{In this course}
Kernel-split quadrature handles the singular \emph{kernel}; RCIP handles the
singular \emph{density}.  We now build RCIP from the ground up.
\end{block}
\end{frame}

The fine mesh subdivides the panels nearest the corner $n_{\rm sub}$ times.  The

%================================================================
\begin{frame}{Operator split and change of variables}
\small
Split each operator so that the star part keeps only the interactions internal
to $\Gamma^\star$:
\[
  K=K^\star+K^\circ,\qquad
  K^\star_{ij}=
  \begin{cases}K_{ij}, & z_i,z_j\in\Gamma^\star,\\[2pt] 0,&\text{otherwise.}\end{cases}
\]
Introduce a \hl{transformed density} $\tilde\rho$ through the variable
substitution
\[
  \rho=(I+\lambda K^\star)^{-1}\tilde\rho.
\]
Substituting into $(I+\lambda K)\rho=\lambda g$ gives the right-preconditioned
equation
\[
  \tilde\rho+\lambda K^\circ(I+\lambda K^\star)^{-1}\tilde\rho=\lambda g.
\]

\begin{block}{The idea}
The difficult inverse $(I+\lambda K^\star)^{-1}$ is \emph{local} to
$\Gamma^\star$.  The new unknown $\tilde\rho$ is what we will represent on the
coarse mesh.
\end{block}
\end{frame}

%================================================================
\begin{frame}{The transformed density is coarse-mesh smooth}
\small
Read the preconditioned equation as
\[
  \tilde\rho+\lambda\,\underbrace{K^\circ(I+\lambda K^\star)^{-1}}_{\text{action is smooth on inner }\Gamma^\star}\tilde\rho=\lambda g.
\]
Every term of $K^\circ$ has at least one index outside $\Gamma^\star$, so its
action on \emph{any} function is smooth on the two innermost coarse panels; the
data $g$ is panelwise smooth there too.

\vspace{1mm}
By contradiction, $\tilde\rho$ must itself be panelwise smooth on those panels.
It is therefore captured by coarse-to-fine polynomial prolongation $P$:
\[
  \tilde\rho_{\rm fin}=P\,\tilde\rho_{\rm coa},
  \qquad
  g_{\rm fin}=P\,g_{\rm coa}.
\]

\begin{block}{Key distinction}
Interpolation acts on the smooth $\tilde\rho$, never on the singular $\rho$.
\end{block}
\end{frame}

%================================================================
\begin{frame}{Prolongation and weighted prolongation}
\small
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.54\textwidth}
$P$ interpolates panelwise degree-15 polynomials from the coarse grid to the
fine grid.  With diagonal matrices $W_{\rm coa}$, $W_{\rm fin}$ holding the
quadrature weights, the weighted prolongation is
\[
  P_W=W_{\rm fin}\,P\,W_{\rm coa}^{-1}.
\]
\begin{itemize}
  \item $P$ and $P_W$ share one sparsity pattern.
  \item Both equal the identity away from $\Gamma^\star$; only one full
  $(4+2n_{\rm sub})16\times64$ block is nontrivial.
\end{itemize}
\end{column}
\begin{column}{0.42\textwidth}
\begin{center}
\includegraphics[width=\textwidth]{day2_morning_figures/rcip_tutorial/pbc.pdf}

\vspace{1.5mm}
\includegraphics[height=0.24\textheight]{day2_morning_figures/rcip_tutorial/pbcpatt.pdf}
\end{center}
\end{column}
\end{columns}
\end{frame}

%================================================================
\begin{frame}{An exact restriction identity}
\small
The weighted transpose $P_W^T$ restricts fine-grid data to the coarse grid
\emph{exactly}:
\[
  P_W^T P=I_{\rm coa}.
\]
\textbf{Why.}  Let $f_{\rm coa}$ and $g_{\rm coa}$ discretize any panelwise
degree-15 polynomials.  Composite 16-point Gauss--Legendre integrates products
of degree $31$ exactly, so the discrete inner product is mesh-independent:
\[
  f_{\rm coa}^T W_{\rm coa} g_{\rm coa}
  =(Pf_{\rm coa})^T W_{\rm fin}(Pg_{\rm coa})
  =f_{\rm coa}^T\,P^T W_{\rm fin} P\,g_{\rm coa}.
\]
Since this holds for a full set of such vectors,
\[
  W_{\rm coa}=P^T W_{\rm fin}P
  \quad\Longrightarrow\quad
  P_W^T P=W_{\rm coa}^{-1}P^T W_{\rm fin}P=I_{\rm coa}.
\]
This exact identity is what makes the compression lossless for data the coarse
mesh already resolves.
\end{frame}

%================================================================
\begin{frame}{Compressing the fine-grid equation}
\small
Two ingredients.  The discrete change of variables on the fine grid,
\[
  \rho_{\rm fin}=(I_{\rm fin}+\lambda K^\star_{\rm fin})^{-1}P\,\tilde\rho_{\rm coa},
\]
and the low-rank decomposition of the smooth part (accurate to machine
precision),
\[
  K^\circ_{\rm fin}=P\,K^\circ_{\rm coa}\,P_W^T .
\]
Insert both into $(I_{\rm fin}+\lambda K_{\rm fin})\rho_{\rm fin}=\lambda P g_{\rm coa}$.
The star part cancels the inverse, leaving
\[
  P\tilde\rho_{\rm coa}
  +\lambda P K^\circ_{\rm coa}P_W^T(I_{\rm fin}+\lambda K^\star_{\rm fin})^{-1}P\tilde\rho_{\rm coa}
  =\lambda P g_{\rm coa}.
\]
Apply $P_W^T$ on the left and use $P_W^T P=I_{\rm coa}$:
\[
  \boxed{\,(I_{\rm coa}+\lambda K^\circ_{\rm coa}R)\,\tilde\rho_{\rm coa}=\lambda g_{\rm coa}\,}
  \qquad
  R=P_W^T(I_{\rm fin}+\lambda K^\star_{\rm fin})^{-1}P .
\]
\end{frame}

%================================================================
\begin{frame}{The compressed inverse $R$}
\small
\[
  (I_{\rm coa}+\lambda K^\circ_{\rm coa}R)\tilde\rho_{\rm coa}=\lambda g_{\rm coa},
  \qquad
  R=P_W^T(I_{\rm fin}+\lambda K^\star_{\rm fin})^{-1}P .
\]
\begin{itemize}
  \item The global system lives on the coarse grid --- $16 n_{\rm pan}$
  unknowns, the same size as if there were no corner.
  \item $K^\star=0$ away from $\Gamma^\star$, so $R=I$ there.  For one corner,
  $R$ is the identity plus a single full $64\times64$ block.
  \item The fine mesh has vanished from the global solve; it survives only inside
  $R$.  Compression changes no discretization error --- no information is lost.
\end{itemize}
\begin{block}{Remaining problem}
The formula for $R$ still inverts the large fine-grid matrix
$I_{\rm fin}+\lambda K^\star_{\rm fin}$, which is itself costly and unstable.  We
replace it by a recursion.
\end{block}
\end{frame}

%================================================================
\begin{frame}{Nested meshes: types \textsf{a}, \textsf{b}, \textsf{c}}
\small
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.54\textwidth}
\begin{center}
\includegraphics[width=\textwidth]{day2_morning_figures/rcip_hj2021/threemeshes.pdf}
\end{center}
\end{column}
\begin{column}{0.42\textwidth}
Nested subsets
$\Gamma_1^\star\subset\cdots\subset\Gamma_{n_{\rm sub}}^\star=\Gamma^\star$, one
per \emph{level} $i$.  On each $\Gamma_i^\star$:
\begin{itemize}
  \item type \textsf{a}: the fine mesh restricted, $4+2i$ panels;
  \item type \textsf{b}: six panels;
  \item type \textsf{c}: four panels.
\end{itemize}
Type \textsf{a} $=$ type \textsf{b} at $i=1$; type \textsf{c} is the coarse
$\Gamma^\star$ at $i=n_{\rm sub}$.
\end{column}
\end{columns}

\vspace{1mm}
Each recursion step compresses a six-panel (type \textsf{b}) problem back to a
four-panel (type \textsf{c}) block.  With 16 nodes per panel it handles a
$96\times96$ matrix and returns a $64\times64$ block --- sizes that never grow
with $n_{\rm sub}$.
\end{frame}

%================================================================
\begin{frame}{The four-to-six-panel prolongations}
\small
$P_{\rm bc}$ interpolates degree-15 polynomials from a 64-point four-panel grid
to a 96-point six-panel grid; $P_{W{\rm bc}}$ is its weighted partner.  From the
16-point nodes and weights $(T_{16},W_{16})$ on $[-1,1]$,
\[
  T_{32}=\tfrac12\!\begin{bmatrix}T_{16}-1\\ T_{16}+1\end{bmatrix},\quad
  W_{32}=\tfrac12\!\begin{bmatrix}W_{16}\\ W_{16}\end{bmatrix},\quad
  A_{jk}=T_{16,j}^{\,k-1},\quad AA_{jk}=T_{32,j}^{\,k-1},
\]
\[
  IP=AA\,A^{-1},\qquad
  IP_W=IP\odot\bigl(W_{32}\,(1/W_{16})^T\bigr),
\]
\[
  P_{\rm bc}=\operatorname{blkdiag}(I,IP,IP,I),\qquad
  P_{W{\rm bc}}=\operatorname{blkdiag}(I,IP_W,IP_W,I).
\]
Only the two middle panels are refined; the outer panels pass through unchanged.
\end{frame}

%================================================================
\begin{frame}{The forward recursion for $R$}
\small
Let $R_i$ be the $64\times64$ block after $i$ levels; $R_{n_{\rm sub}}$ is the
nontrivial block of $R$.  It obeys
\[
  R_i=P_{W{\rm bc}}^T
  \Bigl(\mathbb{F}\{R_{i-1}^{-1}\}+I_{\rm b}^\circ+\lambda K^\circ_{i{\rm b}}\Bigr)^{-1}
  P_{\rm bc},
  \qquad i=1,\ldots,n_{\rm sub},
\]
initialized by
\[
  \mathbb{F}\{R_0^{-1}\}=I_{\rm b}^\star+\lambda K^\star_{1{\rm b}}.
\]
Here $K_{i{\rm b}}=K^\star_{i{\rm b}}+K^\circ_{i{\rm b}}$ is the operator on the
type-\textsf{b} mesh at level $i$, and $\mathbb{F}\{\cdot\}$ zero-pads a
$64\times64$ block into the $96\times96$ type-\textsf{b} format (a frame of width
$16$).  The initializer makes the first step collapse to
\[
  R_1=P_{W{\rm bc}}^T(I_{\rm b}+\lambda K_{1{\rm b}})^{-1}P_{\rm bc}.
\]
The number of recursion steps equals the number $n_{\rm sub}$ of subdivisions.
\end{frame}

%================================================================
\begin{frame}{Where the recursion comes from}
\small
Define $R_i$ through the type-\textsf{a} (fully refined) mesh on $\Gamma_i^\star$:
\[
  R_i=P_{Wi{\rm ac}}^T(I_{i{\rm a}}+\lambda K_{i{\rm a}})^{-1}P_{i{\rm ac}}.
\]
At $i=n_{\rm sub}$ this is exactly the $64\times64$ block of $R$.  Split the
type-\textsf{a} operator so its star part is the previous level, lifted, and
represent the rest through the type-\textsf{b} mesh:
\[
  K_{i{\rm a}}=\mathbb{F}\{K_{(i-1){\rm a}}\}+K^\circ_{i{\rm a}},
  \qquad
  K^\circ_{i{\rm a}}=P_{i{\rm ab}}K^\circ_{i{\rm b}}P_{Wi{\rm ab}}^T .
\]
The prolongations compose, $P_{i{\rm ac}}=P_{i{\rm ab}}P_{\rm bc}$ and
$P_{Wi{\rm ac}}=P_{Wi{\rm ab}}P_{W{\rm bc}}$, so
\[
  R_i=P_{W{\rm bc}}^T P_{Wi{\rm ab}}^T
  \Bigl(I_{i{\rm a}}+\mathbb{F}\{\lambda K_{(i-1){\rm a}}\}
  +\lambda P_{i{\rm ab}}K^\circ_{i{\rm b}}P_{Wi{\rm ab}}^T\Bigr)^{-1}
  P_{i{\rm ab}}P_{\rm bc}.
\]
\end{frame}

%================================================================
\begin{frame}{Collapsing the intermediate mesh}
\small
Expand $(A+B)^{-1}$ with $A=I_{i{\rm a}}+\mathbb{F}\{\lambda K_{(i-1){\rm a}}\}$
and $B=\lambda P_{i{\rm ab}}K^\circ_{i{\rm b}}P_{Wi{\rm ab}}^T$, multiply on the
left by $P_{Wi{\rm ab}}^T$ and on the right by $P_{i{\rm ab}}$, and resum in
closed form:
\[
  R_i=P_{W{\rm bc}}^T
  \Bigl[\bigl(P_{Wi{\rm ab}}^T(I_{i{\rm a}}+\mathbb{F}\{\lambda K_{(i-1){\rm a}}\})^{-1}P_{i{\rm ab}}\bigr)^{-1}
  +\lambda K^\circ_{i{\rm b}}\Bigr]^{-1}P_{\rm bc}.
\]
Zero-padding the definition of $R_{i-1}$ and using
$(P_{Wi{\rm ab}}^T P_{i{\rm ab}})^\circ=I_{\rm b}^\circ$ identifies the old term:
\[
  P_{Wi{\rm ab}}^T(I_{i{\rm a}}+\mathbb{F}\{\lambda K_{(i-1){\rm a}}\})^{-1}P_{i{\rm ab}}
  =\mathbb{F}\{R_{i-1}\}+I_{\rm b}^\circ,
\]
a block-diagonal matrix whose inverse is
$\mathbb{F}\{R_{i-1}^{-1}\}+I_{\rm b}^\circ$.  Substituting recovers the
recursion --- the type-\textsf{a} mesh has disappeared.
\end{frame}

%================================================================
\begin{frame}{Schur--Banachiewicz speedup}
\small
Each step inverts a $96\times96$ matrix, but its inner $64\times64$ block is
already known --- it is $R_{i-1}$.  Partition
\[
  M=\begin{bmatrix}A^{-1}&U\\ V&D\end{bmatrix},\qquad
  A=R_{i-1},\qquad S=D-VAU,
\]
where $U,V,D$ are the remaining blocks of
$I_{\rm b}^\circ+\lambda K^\circ_{i{\rm b}}$.  The Schur--Banachiewicz inverse
formula gives
\[
  M^{-1}=
  \begin{bmatrix}
  A+AUS^{-1}VA & -AUS^{-1}\\[2pt]
  -S^{-1}VA & S^{-1}
  \end{bmatrix}.
\]
Only the small Schur complement $S$ is inverted anew at each level, so building
$R$ costs $\cO(n_{\rm sub})$.  The forward recursion is, in spirit, the upward
pass of a fast multipole method.
\end{frame}

%================================================================
\begin{frame}{After the solve: the weight-corrected density}
\small
Solve the coarse system for $\tilde\rho_{\rm coa}$, then form the
\hl{weight-corrected density}
\[
  \hat\rho_{\rm coa}=R\,\tilde\rho_{\rm coa}.
\]
Rewriting the compressed equation in terms of $\hat\rho_{\rm coa}$,
\[
  (R^{-1}+\lambda K^\circ_{\rm coa})\,\hat\rho_{\rm coa}=\lambda g_{\rm coa},
\]
resembles the original coarse discretization, so $\hat\rho_{\rm coa}$ acts like
pointwise values of $\rho$ carrying weight corrections for Gauss--Legendre
integration.  For any smooth $f$,
\[
  \int_\Gamma f(r)\rho(r)\dd\ell
  \approx\sum_j f(r_{{\rm coa},j})\,\hat\rho_{{\rm coa},j}\,w_{{\rm coa},j}.
\]
Functionals such as the polarizability $q$ come directly from
$\hat\rho_{\rm coa}$ --- the fine-grid density is never formed.
\end{frame}

%================================================================
\begin{frame}{Optional: reconstructing the fine density}
\small
When $\rho_{\rm fin}$ itself is needed --- for instance, to evaluate the field
close to the corner --- it is recovered by running the recursion backwards:
\[
  \vec\rho_{{\rm coa},i}=
  \Bigl[I_{\rm b}-\lambda K^\circ_{i{\rm b}}
  \bigl(\mathbb{F}\{R_{i-1}^{-1}\}+I_{\rm b}^\circ+\lambda K^\circ_{i{\rm b}}\bigr)^{-1}\Bigr]
  P_{\rm bc}\,\tilde\rho_{{\rm coa},i},
  \quad i=n_{\rm sub},\ldots,1.
\]
\begin{itemize}
  \item Each backward step reuses the matrices $K^\circ_{i{\rm b}}$ and $R_i$
  built in the forward recursion.
  \item Interrupting at level $j$ gives a weight-corrected density on a mesh
  refined $n_{\rm sub}-j+1$ times --- exactly what close evaluation needs.
  \item Outside $\Gamma^\star$, $\rho_{\rm fin}$ coincides with
  $\tilde\rho_{\rm coa}$.
\end{itemize}
\end{frame}

%================================================================
\begin{frame}{Why RCIP works so well}
\small
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.5\textwidth}
\textbf{Properties}
\begin{itemize}
  \item The global solve stays on the coarse mesh, independent of $n_{\rm sub}$.
  \item Building $R$ costs $\cO(n_{\rm sub})$, and $n_{\rm sub}$ may exceed
  $100$.
  \item Conditioning does not degrade; close to machine precision is reached.
  \item Only the \emph{locations} of the singular points are needed --- not the
  form of the singularity.
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{How the pieces fit}
\begin{itemize}
  \item Kernel-split quadrature builds the accurate $K^\star$ and $K^\circ$
  blocks.
  \item RCIP compresses the local inverse and resolves the singular density.
  \item Both compose with the FMM for the far field.
\end{itemize}
\end{column}
\end{columns}

\vspace{3mm}
\begin{center}
\hl{Quadrature controls the integration error; RCIP controls the resolution of the unknown.}
\end{center}
\end{frame}
